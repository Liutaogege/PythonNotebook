// 背包问题
#include <stdio.h>
# define MaxSize 100

int W[MaxSize];
// S表示背包的容量，n表示n件物品，每件物品的重量放在W[1...n]中
int S, n;

int cal_res(int S, int n) {
    if (S==0) return 1;
    if (S<0 || (S>0 && n<1)) return 0;
    if (cal_res(S-W[n], n-1)) {
        printf("%d", W[n]);
        return 1;
    }
    else {
        return cal_res(S, n-1);
    }
}

int main() {
    int S0,n0;
    scanf("%d %d", &S0, &n0);
    for (int i=0; i<n0; i++) {
        W[i] = i+1;
    }
    if(cal_res(S0, n0)) printf("success");
    else printf("flase");
}
